  <script>
    let uname = '', pass = '';

    function showPopup() {
      uname = document.getElementById('username').value.trim();
      pass = document.getElementById('password').value.trim();

      if (!uname || !pass) {
        alert("Please enter both fields");
        return;
      }

      document.getElementById('popupText').innerHTML = `
        Confirm your login:<br><br>
        <strong>Username:</strong> ${uname}<br>
        <strong>Password:</strong> ${pass}
      `;
      document.getElementById('popup').style.display = 'block';
    }

    function hidePopup() {
      document.getElementById('popup').style.display = 'none';
    }

    function login() {
      // Just simulating login. You can save to Firebase later
      alert("Login success! Redirecting...");
      window.location.href = "https://instagram.com/bravo_momanyi";
    }
